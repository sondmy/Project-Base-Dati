Table Stati_Esistenza {
  id_stato int [pk]
  nome varchar [not null]        // "in via di estinzione", "vulnerabile", ecc.
  descrizione varchar
}

Table Tipi_Animale {
  id_tipo int [pk]
  nome varchar [not null]        // "mammifero", "rettile", ecc.
  descrizione varchar
}

Table Habitat {
  id_habitat int [pk]
  nome varchar [not null]        // "savana", "foresta tropicale", ecc.
  descrizione varchar
}

Table Tipi_Cibo {
  id_tipo_cibo int [pk]
  nome varchar [not null]        // "carne", "verdure", "pesce", "frutta", ecc.
  descrizione varchar
}
Table Tipi_Area {
  id_tipo_area  int     [pk]
  nome          varchar [not null]   // "tropicale", "savana", "ghiacciaio", ecc.
  descrizione   varchar
}

Table Tipi_Recinto {
  id_tipo_recinto int   [pk]
  nome            varchar [not null] // "voliera", "acquario", "terrario", ecc.
  descrizione     varchar
}

Table Tipi_Fornitura {
  id_tipo_fornitura int [pk]
  nome              varchar [not null]
  descrizione       varchar
}

Table Tipi_Mansione {
  id_tipo_mansione   int     [pk]
  nome          varchar [not null] // Ruoli/mansioni del personale (veterinario, manutentore, guida, contabilità, ecc.)
  descrizione   varchar
}

Table _Specie {
  id_specie         int    [ref: > Specie.id_specie, not null]
  id_tipo_cibo      int    [ref: > Tipi_Cibo.id_tipo_cibo, not null]
  quantita_kg_giorno double [not null]
  indexes {
    (id_specie, id_tipo_cibo) [pk]
  }
}

Table Specie {
  id_specie int [pk]
  nome_scientifico varchar [not null]
  nome_comune varchar [not null]
  id_stato_esistenza int [ref: > Stati_Esistenza.id_stato]
  id_tipo_animale int [ref: > Tipi_Animale.id_tipo]
  id_habitat int [ref: > Habitat.id_habitat]
}

Table Aree {
  id_area       int     [pk]
  nome          varchar [not null]
  metratura     int
  id_tipo_area  int     [ref: > Tipi_Area.id_tipo_area, not null]
}

Table Recinti {
  id_recinto        int [pk]
  id_tipo_recinto   int [ref: > Tipi_Recinto.id_tipo_recinto, not null]
  id_area           int [ref: > Aree.id_area, not null]
}

Table Animali {
  id_animale int [pk]
  nome varchar [not null]
  vivo bool [not null]
  data_nascita date
  data_arrivo date
  data_uscita date
  id_specie int [ref: > Specie.id_specie, not null]
  id_recinto int [ref: > Recinti.id_recinto, not null]
}
Table Storici_Stipendo{
  id_storico int [pk]
  id_dipendente int [ref: > Dipendenti.id_dipendente]
  prezzo_orario double
  data_inizio date
  data_fine date 
}
Table Dipendenti {
  id_dipendente int [pk]
  codice_fiscale varchar [unique, not null]
  nome varchar [not null]
  cognome varchar [not null]
  compleanno date
  id_mansione int [ref: > Tipi_Mansione.id_tipo_mansione, not null]
  data_assunzione date [not null]
  prezzo_orario double [not null]
}

Table Turni {
  id_turno int [pk]
  ora_inizio datetime [not null]
  ora_fine datetime [not null]
  id_dipendente int [ref: > Dipendenti.id_dipendente, not null]
  id_area int [ref: > Aree.id_area, not null]
}

Table Utenti {
  id_utente int [pk]
  email varchar [unique, not null]
  password_hash varchar [not null]
  data_registrazione date
  id_dipendente int [ref: > Dipendenti.id_dipendente]  // nullable: non tutti gli utenti sono dipendenti
}

Table Visite_Mediche {
  id_visita int [pk]
  peso double                        // storico, ricavato in VIEW per l'ultimo
  diagnosi varchar
  note_trattamento varchar
  data_visita date [not null]
  data_fine date
  id_animale int [ref: > Animali.id_animale, not null]
  id_veterinario int [ref: > Dipendenti.id_dipendente, not null]
}

Table Tipi_Biglietti {
  id_biglietto int [pk]
  nome varchar [not null]
  descrizione varchar
  prezzo double [not null]
  attivo bool [not null]
}

Table Scontrini {
  id_scontrino int [pk]
  data_acquisto datetime [not null]
  nome_gruppo varchar
  num_persone int
  id_utente int [ref: > Utenti.id_utente, not null]
}

Table Dettaglio_Scontrini {
  id_dettaglio int [pk]
  id_scontrino int [ref: > Scontrini.id_scontrino, not null]
  id_biglietto int [ref: > Tipi_Biglietti.id_biglietto, not null]
  quantita int [not null]
  prezzo_pagato_biglietto double [not null]  // storico: prezzo al momento dell'acquisto
}

Table Categorie_Transazioni {
  id_categoria int [pk]
  nome varchar [not null]
  descrizione varchar
  tipo enum [not null]   // 'ENTRATA' | 'USCITA' (evita categorie usate nel verso sbagliato)
}

Table Fornitori {
  id_fornitore int [pk]
  nome_azienda varchar [not null]
  descrizione varchar
  indirizzo varchar
  iban varchar
}

Table Transazioni {
  id_transazione int [pk]
  tipo enum [not null]               // 'ENTRATA' | 'USCITA'
  importo double [not null]          // sempre positivo
  data datetime [not null]
  descrizione varchar
  id_categoria int [ref: > Categorie_Transazioni.id_categoria, not null]
  id_utente int [ref: > Utenti.id_utente, not null]
  id_fornitore int [ref: > Fornitori.id_fornitore]  // nullable: solo per uscite verso fornitori
  id_scontrino int [ref: > Scontrini.id_scontrino]  // nullable: solo per entrate da biglietteria
}
