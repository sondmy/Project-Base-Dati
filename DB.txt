Table Animali {
  id_animale int [pk]
  nome varchar
  vivo bool
  data_nascita date
  data_arrivo date
  nome_scientifico varchar [ref: > Specie.nome_scientifico]
  id_recinto int [ref: > Recinti.id_recinto]
}

Table Specie {
  nome_scientifico varchar [pk]
  stato_esistenza varchar
  tipo_animale varchar
  nome_comune varchar
  habitat varchar
}

Table Aree {
  id_area int [pk]
  nome varchar
  metratura int
  tipo varchar
}

Table Recinti {
  id_recinto int [pk]
  tipo_recinto varchar
  id_area int [ref: > Aree.id_area]
}

Table Dipendenti {
  codice_fiscale varchar [pk]
  nome varchar
  cognome varchar
  anno_nascita date
  ruolo int [ref: > Ruoli.id_ruolo]
  data_assunzione date
  prezzo_orario double
}

Table Diete {
  id_dieta int [pk]
  nome_scientifico varchar [ref: > Specie.nome_scientifico]
  tipo_cibo varchar
  quantita double
}

Table Tipo_Biglietti {
  id_biglietto int [pk]
  nome varchar
  descrizione varchar
  prezzo double
  attivo bool
}
Table Dettaglio_scontrini {
  id_dettaglio int [pk]
  id_biglietto int [ref: > Tipo_Biglietti.id_biglietto]
  scontrino int [ref: > Scontrini.id_scontrino]
  quantita int
  prezzo_pagato_biglietto double
}
Table Scontrini {
  id_scontrino int [pk]
  data_acquisto date
  prezzo_totale double
  nome_gruppo varchar
  num_persone int
}
Table Entrate {
  id_entrata int [pk]
  categoria_entrata int [ref: > Categorie.id_categoria]
  data_entrata date
  importo double
  descrizione varchar
  id_utente int [ref: > Utenti.id_utente] 
  // automatico 
}
Table Categorie{
  id_categoria int [pk]
  nome varchar
  descrizione varchar
  tipo_entrata_uscita bool
}
Table Uscite {
  id_uscite int [pk]
  categoria_uscite int [ref: > Categorie.id_categoria]
  data_uscite date
  importo double
  descrizione varchar
  id_utente int [ref: > Utenti.id_utente]
}

Table Visite_Mediche {
  id_trattamento int [pk]
  peso double
  diagnosi varchar
  note_trattamento varchar
  data_visita date
  data_fine date
  id_animale int [ref: > Animali.id_animale]
  id_veterinario varchar [ref: > Dipendenti.codice_fiscale] 
}

Table Utenti {
  id_utente int [pk]
  email varchar
  password varchar
  data_registrazione date
  id_ruolo int [ref: > Ruoli.id_ruolo]
  codice_fiscale varchar [ref: > Dipendenti.codice_fiscale]
}
Table Turni {
  id_lavoro int [pk]
  ora_inizio datetime
  ora_fine datetime
  //data ricavabile da datetime sopra
  codice_fiscale varchar [ref: > Dipendenti.codice_fiscale]
  posto int [ref: > Aree.id_area]
}
Table Ruoli {
  id_ruolo int [pk]
  nome varchar
  descrizione varchar
}
Table Spese {
  id_acquisto int [pk]
  id_utente int [ref: > Utenti.id_utente]
}
Table Permessi {
  id_permesso int [pk]
  codice varchar        // es. "ordina_medicinali", "scrivi_scontrino"
  descrizione varchar
  tabella_target varchar // es. "Uscite", "Scontrino"
}

Table Ruoli_Permesso {
  id_ruolo int [ref: > Ruoli.id_ruolo]
  id_permesso int [ref: > Permessi.id_permesso]
  
  indexes {
    (id_ruolo, id_permesso) [pk]
  }
}
